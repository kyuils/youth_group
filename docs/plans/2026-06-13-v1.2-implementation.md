```js
    function recentSunday() {
      const iso = todayISO();
      const d = new Date(iso + 'T00:00:00');
      return shiftDays(iso, -d.getDay());
    }

    // 'YYYY-MM-DD' → '6월 2주 · 6/14 주일' (그 달의 N번째 주일).
    function weekLabel(iso) {
      const d = new Date(iso + 'T00:00:00');
      if (isNaN(d.getTime())) return iso;
      const month = d.getMonth() + 1;
      const nth = Math.floor((d.getDate() - 1) / 7) + 1;
      return month + '월 ' + nth + '주 · ' + month + '/' + d.getDate() + ' 주일';
    }
```

```jsx
      data.rows.forEach(function(r){ if (r.date === date) byId[r.studentId] = r; });
      const groupOf = function(code){
        return data.students.filter(function(s){ return byId[s.id] && byId[s.id].status === code; });
      };
      const presentGroups = PRESENT_CODES
        .map(function(code){ return { code: code, students: groupOf(code) }; })
        .filter(function(g){ return g.students.length > 0; });
      const etcStudents = groupOf('ETC');
      const absentStudents = groupOf('ABSENT');
      const unmarked = data.students.filter(function(s){ return !byId[s.id]; });
      const presentTotal = presentGroups.reduce(function(a,g){ return a + g.students.length; }, 0) + etcStudents.length;

      const contactChips = function(s){
        const phone = String(s['연락처'] || '').replace(/[^0-9+]/g, '');
        if (!phone) return null;
        return (
          <span style={{display:'inline-flex',gap:6,marginLeft:8}}>
            <a href={'tel:' + phone} className="link-chip" aria-label="전화 걸기">📞</a>
            <a href={'sms:' + phone} className="link-chip" aria-label="문자 보내기">💬</a>
          </span>
        );
      };

      return (
        <>
          <div className="seg-bar">
            <button className={seg === 'week' ? 'active' : ''} onClick={() => setSeg('week')}>주별 명단</button>
            <button className={seg === 'stats' ? 'active' : ''} onClick={() => setSeg('stats')}>통계</button>
          </div>

          {seg === 'stats' && <StatsView idToken={idToken} toast={toast} />}

          {seg === 'week' && (
            <div className="weekly-body">
              <div className="weekly-group">
                <div className="weekly-head">✅ 출석 {presentTotal}</div>
                {presentGroups.map(function(g){
                  return (
                    <div key={g.code} className="weekly-line">
                      <span className={'sum-item ' + g.code}>{labelOf(g.code)}</span>
                      <span className="weekly-names">{g.students.map(function(s){ return s['이름']; }).join(', ')}</span>
                    </div>
                  );
                })}
                {etcStudents.map(function(s){
                  return (
                    <div key={s.id} className="weekly-line">
                      <span className="sum-item ETC">기타</span>
                      <span className="weekly-names">{s['이름']}{byId[s.id].etcText ? ' (' + byId[s.id].etcText + ')' : ''}</span>
                    </div>
                  );
                })}
                {presentTotal === 0 && <div className="weekly-empty">이 주일의 출석 기록이 없습니다</div>}
              </div>

              <div className="weekly-group">
                <div className="weekly-head">❌ 결석 {absentStudents.length}</div>
                {absentStudents.map(function(s){
                  return (
                    <div key={s.id} className="weekly-line">
                      <span className="weekly-names">
                        {s['이름']}{byId[s.id].etcText ? ' (' + byId[s.id].etcText + ')' : ''}
                      </span>
                      {contactChips(s)}
                    </div>
                  );
                })}
                {absentStudents.length === 0 && <div className="weekly-empty">결석한 학생이 없습니다 🎉</div>}
              </div>

              <div className="weekly-group">
                <div className="weekly-head">⬜ 미체크 {unmarked.length}</div>
                {unmarked.length > 0 && (
                  <div className="weekly-line">
                    <span className="weekly-names">{unmarked.map(function(s){ return s['이름']; }).join(', ')}</span>
                  </div>
                )}
              </div>
            </div>
          )}
        </>
      );
    }
```

```js
      const saveAttendance = (studentId, status, etcText) => {
        const key = date + '|' + studentId;
        let prev;
        setAtt((m) => { prev = m[key]; return { ...m, [key]: status }; });
        const rollback = () => {
          setAtt((m) => {
            if (m[key] !== status) return m;
            const next = { ...m };
            if (prev !== undefined) next[key] = prev; else delete next[key];
            return next;
          });
          setAttErr((e) => ({ ...e, [key]: { status, etcText } }));
        };
        const clearErr = () => setAttErr((e) => {
          if (!e[key]) return e;
          const next = { ...e }; delete next[key]; return next;
        });
        callApi('setAttendance', { idToken, date, studentId, status, etcText })
          .then((r) => {
            if (r.ok) { clearErr(); }
            else { rollback(); toast.show('저장 실패: ' + (r.code || '')); }
          })
          .catch(() => { rollback(); toast.show('네트워크 오류 — ⟳로 재시도하세요'); });
      };
```

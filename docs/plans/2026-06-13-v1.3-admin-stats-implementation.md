    // 관리자 통계: 전체 요약 + 주별 추이 + 반별 + 학년별 (올해 고정, 클라이언트 집계)
    function AdminStats({ byClass, attRows }) {
      const classNames = Object.keys(byClass);
      const allStudents = classNames.reduce(function(acc, c){ return acc.concat(byClass[c] || []); }, []);
      const dates = Array.from(new Set(attRows.map(function(r){ return r.date; }))).sort();
      const totalDates = dates.length;

      if (allStudents.length === 0) return <div className="loading">학생이 없습니다.</div>;
      if (totalDates === 0) return <div className="loading">올해 출석 기록이 아직 없습니다.</div>;

      const presentRows = attRows.filter(function(r){ return PRESENT_CODES.indexOf(r.status) >= 0; });

      // 전체 요약
      const overallTotal = allStudents.length * totalDates;
      const overallPct = Math.min(100, Math.round((presentRows.length / overallTotal) * 100));
      const latestDate = dates[dates.length - 1];
      const latestPresent = presentRows.filter(function(r){ return r.date === latestDate; }).length;
      const latestPct = Math.min(100, Math.round((latestPresent / allStudents.length) * 100));

      // 주별 추이 (최근 12주)
      const trendDates = dates.slice(-12);
      const trendMax = Math.max.apply(null, trendDates.map(function(d){
        return presentRows.filter(function(r){ return r.date === d; }).length;
      }).concat([1]));

      return (
        <>
          <div className="stats-summary">
            <div className="stats-summary-title">올해 전체 평균 출석률</div>
            <div className="stats-summary-row">
              <span className="big">{overallPct}%</span>
              <span className="sub">({totalDates}회 예배 · 학생 {allStudents.length}명)</span>
            </div>
            <div className="stats-summary-meta">
              최근 예배({latestDate}): <strong>{latestPresent}/{allStudents.length}명</strong> 출석 · {latestPct}%
            </div>
          </div>

          <div className="admin-section">
            <h4>주별 출석 추이 (최근 {trendDates.length}주)</h4>
            <div className="trend-chart">
              {trendDates.map(function(d){
                const n = presentRows.filter(function(r){ return r.date === d; }).length;
                return (
                  <div key={d} className="trend-col" title={d + ' · ' + n + '명'}>
                    <span className="trend-num">{n}</span>
                    <div className="trend-fill" style={{height: Math.max(4, Math.round((n / trendMax) * 60)) + 'px'}}></div>
                    <span className="trend-date">{Number(d.slice(5,7)) + '/' + Number(d.slice(8,10))}</span>
                  </div>
                );
              })}
            </div>
          </div>

          <GroupStatSection title="반별 통계" groups={[]} />
          <GroupStatSection title="학년별 통계" groups={[]} />
        </>
      );
    }

    // 그룹(반/학년) 통계 공용 렌더러. groups: [{name, count, present, total, pct}]
    function GroupStatSection({ title, groups }) {
      return (
        <div className="admin-section">
          <h4>{title}</h4>
          {groups.length === 0 && <div style={{fontSize:14,color:'var(--label-alt)'}}>데이터가 없습니다.</div>}
          {groups.map(function(g){
            return (
              <div key={g.name} className="stat-bar-card">
                <div className="stat-bar-head">
                  <span className="stat-bar-name">{g.name} <span style={{fontWeight:400,color:'var(--label-alt)'}}>({g.count}명)</span></span>
                  <span className="stat-bar-pct">{g.pct}%</span>
                </div>
                <div className="stat-bar-track">
                  <div className="stat-bar-fill" style={{width: Math.max(2, g.pct) + '%'}}></div>
                </div>
                <div className="stat-bar-meta">{g.present} / {g.total}회 출석</div>
              </div>
            );
          })}
        </div>
      );
    }
```

      // 학년별: 현재 명단의 학년 기준 (출석 기록엔 학년 없음). 빈 학년 → '기타'.
      const GRADE_ORDER = ['중1','중2','중3','고1','고2','고3'];

      gradeNames.sort(function(a,b){

                 pct: total > 0 ? Math.min(100, Math.round((present / total) * 100)) : 0 };

      const year = String(new Date().getFullYear());
      const ncStudents = (newcomers && newcomers.students) || [];
      const dates = Array.from(new Set(attRows.map(function(r){ return r.date; }))).sort();
      const totalDates = dates.length;

      // 올해 등반 완료자 (graduates 없는 구 GAS 응답과 하위호환)
      const hasGradApi = !!(newcomers && newcomers.graduates);
      const gradsThisYear = ((newcomers && newcomers.graduates) || [])
        .filter(function(g){ return String(g.graduatedOn || '').slice(0, 4) === year; })
        .sort(function(a,b){ return String(b.graduatedOn).localeCompare(String(a.graduatedOn)); });

      const WEEKS = [1, 2, 3, 4];

      return (
        <>
          <div className="admin-section">
            <h4>새가족 출석 현황 (올해 {totalDates}회)</h4>
            {ncStudents.length === 0 && <div style={{fontSize:14,color:'var(--label-alt)'}}>현재 새가족 학생이 없습니다.</div>}
            {ncStudents.map(function(s){
              const mine = attRows.filter(function(r){ return String(r.studentId) === String(s.id); });
              const present = mine.filter(function(r){ return PRESENT_CODES.indexOf(r.status) >= 0; }).length;
              const pct = totalDates > 0 ? Math.min(100, Math.round((present / totalDates) * 100)) : 0;
              return (
                <div key={s.id} className="stat-bar-card">
                  <div className="stat-bar-head">
                    <span className="stat-bar-name">{s['이름']}</span>
                    <span className="stat-bar-pct">{pct}%</span>
                  </div>
                  <div className="stat-bar-track">
                    <div className="stat-bar-fill" style={{width: Math.max(2, pct) + '%'}}></div>
                  </div>
                  <div className="stat-bar-meta">{present} / {totalDates}회 출석 · {s['학년']}</div>
                </div>
              );
            })}
          </div>

          <div className="admin-section">
            <h4>등반 교육 진행 <span style={{fontWeight:400,fontSize:12,color:'var(--label-alt)'}}>(수정은 새가족 탭에서)</span></h4>
            {ncStudents.length === 0 && <div style={{fontSize:14,color:'var(--label-alt)'}}>진행 중인 학생이 없습니다.</div>}
            {ncStudents.map(function(s){
              const pr = s.progress || {};
              return (
                <div key={s.id} style={{display:'flex',alignItems:'center',gap:8,padding:'8px 0',borderBottom:'1px solid var(--fill-normal)'}}>
                  <span style={{flex:1,fontSize:14,fontWeight:500}}>{s['이름']}</span>
                  {WEEKS.map(function(w){
                    const done = !!pr['week' + w];
                    return (
                      <span key={w} className={'grad-week' + (done ? ' done' : '')}
                        title={done ? (pr['week' + w + 'Date'] || '') : ''}>
                        {w}주{done ? ' ✓' : ''}
                      </span>
                    );
                  })}
                </div>
              );
            })}
          </div>

          <div className="admin-section">
            <h4>올해 등반 완료 ({gradsThisYear.length}명)</h4>
            {!hasGradApi && <div style={{fontSize:14,color:'var(--label-alt)'}}>등반 명단은 GAS 재배포 후 표시됩니다.</div>}
            {hasGradApi && gradsThisYear.length === 0 && <div style={{fontSize:14,color:'var(--label-alt)'}}>올해 등반 완료자가 없습니다.</div>}
            {gradsThisYear.map(function(g){
              return (
                <div key={g.id} style={{display:'flex',alignItems:'center',gap:8,padding:'8px 0',borderBottom:'1px solid var(--fill-normal)',fontSize:14}}>
                  <span style={{flex:1,fontWeight:500}}>{g['이름'] || ('학생 ' + g.id)}
                    {g['학년'] ? <span style={{fontWeight:400,color:'var(--label-alt)',marginLeft:6,fontSize:12}}>{g['학년']}</span> : null}
                  </span>
                  <span style={{color:'var(--label-alt)',fontSize:13}}>{g.graduatedOn}</span>
                  <span className="sum-item" style={{fontSize:12}}>{g.graduatedTo} 반</span>
                </div>
              );
            })}
          </div>
        </>
      );
    }
```
